Traffic Light

เพิ่มประสิทธิภาพการจราจรด้วยเทคโนโลยีไฟสัญญาณ

บทคัดย่อ

โครงงานนี้มีวัตถุประสงค์เพื่อออกแบบและสร้างสัญญาณไฟจราจรจำลองโดยใช้ Arduino R4 เป็นตัวควบคุมสลับการทำงานของไฟ LED ทั้งสามสี ได้แก่ สีแดง สีเหลือง และสีเขียวตามลำดับเวลาเหมือนสัญญาณไฟจราจรในชีวิตจริง ระบบที่พัฒนานี้เน้นการตั้งค่าเวลาในการเปลี่ยนสัญญาณไฟอย่างเหมาะสม เพื่อให้การทำงานคล้ายคลึงกับสัญญาณไฟจราจรในเมือง โครงงานนี้มีประโยชน์ต่อการเรียนรู้ด้านการเขียนโปรแกรม การออกแบบวงจร และการประยุกต์ใช้ไมโครคอนโทรลเลอร์ในการควบคุมอุปกรณ์อิเล็กทรอนิกส์ ช่วยส่งเสริมความเข้าใจในการควบคุมสัญญาณไฟและการจัดการจราจรโดยใช้ระบบอัตโนมัติ ซึ่งสามารถต่อยอดการเรียนรู้และพัฒนาไปสู่การใช้งานในระบบควบคุมอื่น ๆ ที่ซับซ้อนขึ้น โครงงานนี้ช่วยฝึกฝนทักษะการแก้ปัญหา การวิเคราะห์ และการออกแบบระบบควบคุม ทั้งยังสนับสนุนให้ผู้เรียนสามารถนำความรู้ที่ได้ไปใช้ในงานด้านการจราจรหรือระบบอัตโนมัติในอนาคต

คุณสมบัติ

การประหยัดพลังงาน

ใช้เทคโนโลยี LED และระบบจัดการพลังงานที่มีประสิทธิภาพ ช่วยลดการใช้พลังงาน

การวิเคราะห์ข้อมูล

เก็บและวิเคราะห์ข้อมูลการจราจรเพื่อปรับปรุงและวางแผนการจราจรในอนาคต

การบำรุงรักษาง่าย

ออกแบบให้สามารถบำรุงรักษาและตรวจสอบระบบได้อย่างสะดวกและรวดเร็ว

// C++ code
//
int green1=6,yellew1=7,green2=8,yellew2=9;
int green3=10,yellew3=11,green4=12,yellew4=13;
int sw=5,blink=4;
int green=5000, yellow=3000, taffic_delay=2000;

void setup()
{
  pinMode(green1, OUTPUT);
  pinMode(yellew1, OUTPUT);
  pinMode(green2, OUTPUT);
  pinMode(yellew2, OUTPUT);
  pinMode(green3, OUTPUT);
  pinMode(yellew3, OUTPUT);
  pinMode(green4, OUTPUT);
  pinMode(yellew4, OUTPUT);
  pinMode(blink, OUTPUT);
  pinMode(sw, INPUT_PULLUP);
  Serial.begin(9600);
}

void loop()
{
  if(!digitalRead(sw)){
    
    while(!digitalRead(sw)){
      //ชุดที่1
      Serial.println('1');
      digitalWrite(green1,HIGH); 
      delay(green);
      digitalWrite(yellew1,HIGH);
      digitalWrite(green1,LOW); 
      delay(yellow); 
      digitalWrite(yellew1,LOW); 
      delay(taffic_delay);
      
      //ชุดที่2
      Serial.println('2');
      digitalWrite(green2,HIGH); 
      delay(green);
      digitalWrite(yellew2,HIGH); 
      digitalWrite(green2,LOW);
      delay(yellow); 
      digitalWrite(yellew2,LOW); 
      delay(taffic_delay);
      
      //ชุดที่3
      Serial.println('3');
      digitalWrite(green3,HIGH); 
      delay(green);
      digitalWrite(yellew3,HIGH); 
      digitalWrite(green3,LOW);
      delay(yellow); 
      digitalWrite(yellew3,LOW); 
      delay(taffic_delay);
      
      //ชุดที่4
      Serial.println('4');
      digitalWrite(green4,HIGH); 
      delay(green);
      digitalWrite(yellew4,HIGH); 
      digitalWrite(green4,LOW); 
      delay(yellow); 
      digitalWrite(yellew4,LOW); 
      delay(taffic_delay);
    }
 
  } else {
    Serial.println('0');
    digitalWrite(blink,HIGH); 
    delay(1000); 
    digitalWrite(blink,LOW);
    delay(1000);  
  }
}
